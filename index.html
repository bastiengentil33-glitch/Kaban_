// ─── HELPERS ─────────────────────────────────────────────────────────────────
function formatDate(d) {
  if (!d) return "—";
  return new Date(d).toLocaleDateString("fr-FR", { day: "2-digit", month: "short", year: "numeric" });
}
function daysLeft(d) {
  if (!d) return null;
  const diff = Math.ceil((new Date(d) - new Date()) / 86400000);
  return diff;
}
function uuid() { return Math.random().toString(36).slice(2); }

// ─── TASK MODAL ───────────────────────────────────────────────────────────────
function TaskModal({ task, onClose, onSave, onDelete }) {
  const [t, setT] = useState(task ? { ...task, checklist: task.checklist.map(c => ({ ...c })) } : {
    id: uuid(), title: "", description: "", status: "todo", priority: "medium",
    color: "#6C63FF", tags: [], assignee: "", startDate: "", dueDate: "", progress: 0,
    checklist: [], reminders: [], subtasks: []
  });
  const [newCheck, setNewCheck] = useState("");
  const [newTag, setNewTag] = useState("");
  const [newReminder, setNewReminder] = useState({ date: "", label: "" });

  const update = (k, v) => setT(prev => ({ ...prev, [k]: v }));
  const addCheck = () => { if (!newCheck.trim()) return; update("checklist", [...t.checklist, { id: uuid(), text: newCheck, done: false }]); setNewCheck(""); };
  const toggleCheck = (id) => update("checklist", t.checklist.map(c => c.id === id ? { ...c, done: !c.done } : c));
  const removeCheck = (id) => update("checklist", t.checklist.filter(c => c.id !== id));
  const addTag = () => { if (!newTag.trim()) return; update("tags", [...(t.tags || []), newTag.trim()]); setNewTag(""); };
  const removeTag = (tag) => update("tags", t.tags.filter(tg => tg !== tag));
  const addReminder = () => { if (!newReminder.date) return; update("reminders", [...t.reminders, { id: uuid(), ...newReminder }]); setNewReminder({ date: "", label: "" }); };
  const checkProgress = Math.round(t.checklist.length ? (t.checklist.filter(c => c.done).length / t.checklist.length) * 100 : 0);

  return (
    <div style={{ position: "fixed", inset: 0, background: "rgba(10,10,20,0.85)", zIndex: 1000, display: "flex", alignItems: "center", justifyContent: "center", padding: 16 }} onClick={onClose}>
      <div onClick={e => e.stopPropagation()} style={{
        background: "#0F1117", border: "1px solid #2A2D3A", borderRadius: 20, width: "min(720px,100%)",
        maxHeight: "90vh", overflowY: "auto", padding: 32, boxShadow: "0 40px 80px rgba(0,0,0,0.6)",
        scrollbarWidth: "thin", scrollbarColor: "#2A2D3A #0F1117"
      }}>
        {/* Header */}
        <div style={{ display: "flex", alignItems: "center", gap: 12, marginBottom: 24 }}>
          <input type="color" value={t.color} onChange={e => update("color", e.target.value)} style={{ width: 40, height: 40, border: "none", borderRadius: 8, cursor: "pointer", background: "none" }} />
          <input value={t.title} onChange={e => update("title", e.target.value)} placeholder="Titre de la tâche..." style={{
            flex: 1, background: "transparent", border: "none", fontSize: 22, fontWeight: 700,
            color: "#F0F0FF", outline: "none", fontFamily: "inherit"
          }} />
          <button onClick={onClose} style={{ background: "none", border: "none", color: "#666", fontSize: 22, cursor: "pointer" }}>✕</button>
        </div>

        {/* Grid fields */}
        <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: 16, marginBottom: 20 }}>
          {[
            { label: "Statut", key: "status", type: "select", options: COLUMNS.map(c => ({ value: c.id, label: c.label })) },
            { label: "Priorité", key: "priority", type: "select", options: Object.entries(PRIORITY_LABELS).map(([v, l]) => ({ value: v, label: l })) },
            { label: "Assigné à", key: "assignee", type: "text" },
            { label: "Progression (%)", key: "progress", type: "number" },
            { label: "Date début", key: "startDate", type: "date" },
            { label: "Date fin", key: "dueDate", type: "date" },
          ].map(f => (
            <div key={f.key}>
              <label style={{ color: "#888", fontSize: 11, textTransform: "uppercase", letterSpacing: 1 }}>{f.label}</label>
              {f.type === "select" ? (
                <select value={t[f.key]} onChange={e => update(f.key, e.target.value)} style={inputStyle}>
                  {f.options.map(o => <option key={o.value} value={o.value}>{o.label}</option>)}
                </select>
              ) : (
                <input type={f.type} value={t[f.key]} onChange={e => update(f.key, f.type === "number" ? +e.target.value : e.target.value)} style={inputStyle} min={f.type === "number" ? 0 : undefined} max={f.type === "number" ? 100 : undefined} />
              )}
            </div>
          ))}
        </div>

        {/* Description */}
        <div style={{ marginBottom: 20 }}>
          <label style={labelStyle}>Description</label>
          <textarea value={t.description} onChange={e => update("description", e.target.value)} rows={3} style={{ ...inputStyle, resize: "vertical", lineHeight: 1.6 }} placeholder="Détails de la tâche..." />
        </div>

        {/* Progress bar */}
        <div style={{ marginBottom: 20 }}>
          <label style={labelStyle}>Progression checklist: {checkProgress}%</label>
          <div style={{ height: 6, background: "#1E2130", borderRadius: 4, overflow: "hidden" }}>
            <div style={{ height: "100%", width: `${checkProgress}%`, background: `linear-gradient(90deg, ${t.color}, ${t.color}cc)`, borderRadius: 4, transition: "width .3s" }} />
          </div>
        </div>

        {/* Tags */}
        <div style={{ marginBottom: 20 }}>
          <label style={labelStyle}>Tags</label>
          <div style={{ display: "flex", gap: 8, flexWrap: "wrap", marginBottom: 8 }}>
            {(t.tags || []).map(tag => (
              <span key={tag} style={{ background: "#1E2130", border: "1px solid #2A2D3A", color: "#AAA", padding: "4px 10px", borderRadius: 20, fontSize: 12, display: "flex", alignItems: "center", gap: 6 }}>
                {tag} <span onClick={() => removeTag(tag)} style={{ cursor: "pointer", color: "#FF6584" }}>×</span>
              </span>
            ))}
          </div>
          <div style={{ display: "flex", gap: 8 }}>
            <input value={newTag} onChange={e => setNewTag(e.target.value)} onKeyDown={e => e.key === "Enter" && addTag()} placeholder="Nouveau tag..." style={{ ...inputStyle, flex: 1 }} />
            <button onClick={addTag} style={btnSmall}>+</button>
          </div>
        </div>

        {/* Checklist */}
        <div style={{ marginBottom: 20 }}>
          <label style={labelStyle}>Checklist ({t.checklist.filter(c => c.done).length}/{t.checklist.length})</label>
          {t.checklist.map(c => (
            <div key={c.id} style={{ display: "flex", alignItems: "center", gap: 10, padding: "6px 0", borderBottom: "1px solid #1E2130" }}>
              <div onClick={() => toggleCheck(c.id)} style={{
                width: 18, height: 18, borderRadius: 5, border: `2px solid ${c.done ? t.color : "#333"}`,
                background: c.done ? t.color : "transparent", cursor: "pointer", display: "flex", alignItems: "center", justifyContent: "center", flexShrink: 0
              }}>{c.done && <span style={{ fontSize: 11, color: "#fff" }}>✓</span>}</div>
              <span style={{ flex: 1, color: c.done ? "#555" : "#DDD", textDecoration: c.done ? "line-through" : "none", fontSize: 14 }}>{c.text}</span>
              <span onClick={() => removeCheck(c.id)} style={{ cursor: "pointer", color: "#FF6584", fontSize: 16 }}>×</span>
            </div>
          ))}
          <div style={{ display: "flex", gap: 8, marginTop: 8 }}>
            <input value={newCheck} onChange={e => setNewCheck(e.target.value)} onKeyDown={e => e.key === "Enter" && addCheck()} placeholder="Ajouter un élément..." style={{ ...inputStyle, flex: 1 }} />
            <button onClick={addCheck} style={btnSmall}>+</button>
          </div>
        </div>

        {/* Reminders */}
        <div style={{ marginBottom: 24 }}>
          <label style={labelStyle}>🔔 Rappels & Notifications</label>
          {t.reminders.map(r => (
            <div key={r.id} style={{ display: "flex", alignItems: "center", gap: 10, padding: "6px 10px", background: "#1A1D2E", borderRadius: 8, marginBottom: 6 }}>
              <span style={{ fontSize: 16 }}>🔔</span>
              <span style={{ flex: 1, color: "#CCC", fontSize: 13 }}>{r.label || "Rappel"} — {formatDate(r.date)}</span>
              <span onClick={() => update("reminders", t.reminders.filter(x => x.id !== r.id))} style={{ cursor: "pointer", color: "#FF6584" }}>×</span>
            </div>
          ))}
          <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr auto", gap: 8, marginTop: 8 }}>
            <input type="datetime-local" value={newReminder.date} onChange={e => setNewReminder(p => ({ ...p, date: e.target.value }))} style={inputStyle} />
            <input value={newReminder.label} onChange={e => setNewReminder(p => ({ ...p, label: e.target.value }))} placeholder="Label..." style={inputStyle} />
            <button onClick={addReminder} style={btnSmall}>+</button>
          </div>
        </div>

        {/* Actions */}
        <div style={{ display: "flex", gap: 12, justifyContent: "flex-end" }}>
          {task && <button onClick={() => { onDelete(task.id); onClose(); }} style={{ ...btnBase, background: "#2A1520", color: "#FF6584", border: "1px solid #FF658440" }}>Supprimer</button>}
          <button onClick={onClose} style={{ ...btnBase, background: "#1E2130", color: "#AAA", border: "1px solid #2A2D3A" }}>Annuler</button>
          <button onClick={() => { onSave(t); onClose(); }} style={{ ...btnBase, background: t.color, color: "#fff", border: "none" }}>💾 Enregistrer</button>
        </div>
      </div>
    </div>
  );
}

const inputStyle = { display: "block", width: "100%", background: "#1A1D2E", border: "1px solid #2A2D3A", borderRadius: 8, padding: "8px 12px", color: "#F0F0FF", fontSize: 14, outline: "none", marginTop: 4, fontFamily: "inherit", boxSizing: "border-box" };
const labelStyle = { color: "#888", fontSize: 11, textTransform: "uppercase", letterSpacing: 1, display: "block", marginBottom: 6 };
const btnSmall = { background: "#6C63FF", border: "none", color: "#fff", borderRadius: 8, padding: "8px 14px", cursor: "pointer", fontSize: 16, fontFamily: "inherit" };
const btnBase = { padding: "10px 20px", borderRadius: 10, cursor: "pointer", fontFamily: "inherit", fontSize: 14, fontWeight: 600 };

// ─── KANBAN VIEW ──────────────────────────────────────────────────────────────
function KanbanView({ tasks, onTaskClick, onAddTask }) {
  const [dragging, setDragging] = useState(null);
  const [over, setOver] = useState(null);

  return (
    <div style={{ display: "flex", gap: 20, padding: "20px 24px", overflowX: "auto", minHeight: "calc(100vh - 160px)" }}>
      {COLUMNS.map(col => {
        const colTasks = tasks.filter(t => t.status === col.id);
        return (
          <div key={col.id}
            onDragOver={e => { e.preventDefault(); setOver(col.id); }}
            onDrop={e => { e.preventDefault(); if (dragging) { onTaskClick({ ...dragging, status: col.id }, true); setDragging(null); setOver(null); } }}
            style={{ minWidth: 300, flex: "0 0 300px" }}>
            {/* Column header */}
            <div style={{ display: "flex", alignItems: "center", justifyContent: "space-between", marginBottom: 14 }}>
              <div style={{ display: "flex", alignItems: "center", gap: 8 }}>
                <div style={{ width: 10, height: 10, borderRadius: "50%", background: col.color, boxShadow: `0 0 8px ${col.color}` }} />
                <span style={{ color: "#DDD", fontWeight: 700, fontSize: 14 }}>{col.label}</span>
                <span style={{ background: "#1E2130", color: "#666", fontSize: 11, padding: "2px 8px", borderRadius: 20 }}>{colTasks.length}</span>
              </div>
              <button onClick={() => onAddTask(col.id)} style={{ background: "none", border: "1px solid #2A2D3A", color: "#666", width: 28, height: 28, borderRadius: 8, cursor: "pointer", fontSize: 16, display: "flex", alignItems: "center", justifyContent: "center" }}>+</button>
            </div>

            {/* Drop zone */}
            <div style={{
              minHeight: 80, borderRadius: 14, padding: 4,
              background: over === col.id ? `${col.color}11` : "transparent",
              border: over === col.id ? `2px dashed ${col.color}55` : "2px dashed transparent",
              transition: "all .2s"
            }}>
              {colTasks.map(task => <KanbanCard key={task.id} task={task} onClick={() => onTaskClick(task)} onDragStart={() => setDragging(task)} />)}
            </div>
          </div>
        );
      })}
    </div>
  );
}

function KanbanCard({ task, onClick, onDragStart }) {
  const dl = daysLeft(task.dueDate);
  const checkDone = task.checklist.filter(c => c.done).length;
  const checkTotal = task.checklist.length;
  return (
    <div draggable onDragStart={onDragStart} onClick={onClick} style={{
      background: "#0D0F1A", border: "1px solid #2A2D3A", borderLeft: `4px solid ${task.color}`,
      borderRadius: 12, padding: 16, marginBottom: 10, cursor: "pointer", transition: "all .2s",
      position: "relative", overflow: "hidden"
    }}
      onMouseEnter={e => { e.currentTarget.style.transform = "translateY(-2px)"; e.currentTarget.style.boxShadow = `0 8px 24px ${task.color}22`; }}
      onMouseLeave={e => { e.currentTarget.style.transform = ""; e.currentTarget.style.boxShadow = ""; }}>
      <div style={{ position: "absolute", top: 0, right: 0, width: 60, height: 60, background: `radial-gradient(circle at top right, ${task.color}15, transparent)`, pointerEvents: "none" }} />
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "flex-start", marginBottom: 8 }}>
        <span style={{ color: "#F0F0FF", fontWeight: 600, fontSize: 14, lineHeight: 1.4, flex: 1 }}>{task.title}</span>
        <span style={{ fontSize: 10, padding: "2px 8px", borderRadius: 12, background: `${PRIORITY_COLORS[task.priority]}22`, color: PRIORITY_COLORS[task.priority], marginLeft: 8, flexShrink: 0 }}>{PRIORITY_LABELS[task.priority]}</span>
      </div>
      {task.description && <p style={{ color: "#666", fontSize: 12, margin: "0 0 10px", lineHeight: 1.5, display: "-webkit-box", WebkitLineClamp: 2, WebkitBoxOrient: "vertical", overflow: "hidden" }}>{task.description}</p>}
      {(task.tags || []).length > 0 && (
        <div style={{ display: "flex", gap: 4, flexWrap: "wrap", marginBottom: 10 }}>
          {task.tags.map(tag => <span key={tag} style={{ fontSize: 10, padding: "2px 8px", borderRadius: 10, background: "#1E2130", color: "#888" }}>{tag}</span>)}
        </div>
      )}
      {checkTotal > 0 && (
        <div style={{ marginBottom: 10 }}>
          <div style={{ display: "flex", justifyContent: "space-between", marginBottom: 4 }}>
            <span style={{ fontSize: 11, color: "#666" }}>☑ {checkDone}/{checkTotal}</span>
            <span style={{ fontSize: 11, color: "#666" }}>{Math.round(checkDone / checkTotal * 100)}%</span>
          </div>
          <div style={{ height: 3, background: "#1E2130", borderRadius: 2 }}>
            <div style={{ height: "100%", width: `${checkDone / checkTotal * 100}%`, background: task.color, borderRadius: 2 }} />
          </div>
        </div>
      )}
      <div style={{ display: "flex", justifyContent: "space-between", alignItems: "center" }}>
        <span style={{ fontSize: 11, color: task.assignee ? "#888" : "#444" }}>👤 {task.assignee || "Non assigné"}</span>
        {task.dueDate && (
          <span style={{ fontSize: 11, color: dl !== null && dl < 3 ? "#FF6584" : "#666" }}>
            {dl !== null && dl < 0 ? "⚠ En retard" : dl !== null && dl <= 3 ? `⏰ ${dl}j restants` : `📅 ${formatDate(task.dueDate)}`}
          </span>
        )}
      </div>
      {task.reminders?.length > 0 && <div style={{ marginTop: 6, fontSize: 11, color: "#FFB347" }}>🔔 {task.reminders.length} rappel{task.reminders.length > 1 ? "s" : ""}</div>}
    </div>
  );
}

// ─── LIST VIEW ────────────────────────────────────────────────────────────────
function ListView({ tasks, onTaskClick }) {
  const [sort, setSort] = useState("dueDate");
  const sorted = [...tasks].sort((a, b) => {
    if (sort === "priority") return ["high", "medium", "low"].indexOf(a.priority) - ["high", "medium", "low"].indexOf(b.priority);
    if (sort === "dueDate") return (a.dueDate || "z") > (b.dueDate || "z") ? 1 : -1;
    if (sort === "title") return a.title.localeCompare(b.title);
    return 0;
  });
  return (
    <div style={{ padding: "20px 24px" }}>
      <div style={{ display: "flex", gap: 8, marginBottom: 16 }}>
        {["title", "dueDate", "priority"].map(s => (
          <button key={s} onClick={() => setSort(s)} style={{ padding: "6px 14px", borderRadius: 8, border: "1px solid #2A2D3A", background: sort === s ? "#6C63FF" : "#1A1D2E", color: sort === s ? "#fff" : "#888", cursor: "pointer", fontSize: 12, fontFamily: "inherit" }}>
            {s === "title" ? "Titre" : s === "dueDate" ? "Échéance" : "Priorité"}
          </button>
        ))}
      </div>
      <div style={{ background: "#0D0F1A", border: "1px solid #2A2D3A", borderRadius: 14, overflow: "hidden" }}>
        <div style={{ display: "grid", gridTemplateColumns: "3fr 1fr 1fr 1fr 1fr 80px", padding: "10px 16px", borderBottom: "1px solid #2A2D3A", background: "#1A1D2E" }}>
          {["Tâche", "Statut", "Priorité", "Assigné", "Échéance", "Avancement"].map(h => (
            <span key={h} style={{ color: "#666", fontSize: 11, textTransform: "uppercase", letterSpacing: 1 }}>{h}</span>
          ))}
        </div>
        {sorted.map(task => {
          const col = COLUMNS.find(c => c.id === task.status);
          const dl = daysLeft(task.dueDate);
          return (
            <div key={task.id} onClick={() => onTaskClick(task)} style={{ display: "grid", gridTemplateColumns: "3fr 1fr 1fr 1fr 1fr 80px", padding: "12px 16px", borderBottom: "1px solid #1A1D2E", cursor: "pointer", transition: "background .15s", alignItems: "center" }}
              onMouseEnter={e => e.currentTarget.style.background = "#1A1D2E"}
              onMouseLeave={e => e.currentTarget.style.background = ""}>
              <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
                <div style={{ width: 8, height: 8, borderRadius: "50%", background: task.color, flexShrink: 0 }} />

// ─── CALENDAR VIEW ────────────────────────────────────────────────────────────
function CalendarView({ tasks, onTaskClick }) {
  const [current, setCurrent] = useState(new Date("2026-06-01"));
  const year = current.getFullYear(), month = current.getMonth();
  const firstDay = new Date(year, month, 1).getDay();
  const daysInMonth = new Date(year, month + 1, 0).getDate();
  const cells = [];
  for (let i = 0; i < (firstDay === 0 ? 6 : firstDay - 1); i++) cells.push(null);
  for (let d = 1; d <= daysInMonth; d++) cells.push(d);
  const monthStr = current.toLocaleDateString("fr-FR", { month: "long", year: "numeric" });
  const prev = () => setCurrent(new Date(year, month - 1, 1));
  const next = () => setCurrent(new Date(year, month + 1, 1));

  return (
    <div style={{ padding: "20px 24px" }}>
      <div style={{ display: "flex", alignItems: "center", gap: 16, marginBottom: 20 }}>
        <button onClick={prev} style={navBtn}>‹</button>
        <h3 style={{ color: "#F0F0FF", margin: 0, fontSize: 18, fontWeight: 700, textTransform: "capitalize" }}>{monthStr}</h3>
        <button onClick={next} style={navBtn}>›</button>
      </div>
      <div style={{ display: "grid", gridTemplateColumns: "repeat(7, 1fr)", gap: 2, marginBottom: 2 }}>
        {["Lun", "Mar", "Mer", "Jeu", "Ven", "Sam", "Dim"].map(d => (
          <div key={d} style={{ textAlign: "center", color: "#555", fontSize: 11, fontWeight: 700, padding: "8px 0", textTransform: "uppercase", letterSpacing: 1 }}>{d}</div>
        ))}
      </div>
      <div style={{ display: "grid", gridTemplateColumns: "repeat(7, 1fr)", gap: 2 }}>
        {cells.map((day, i) => {
          if (!day) return <div key={`e${i}`} />;
          const dateStr = `${year}-${String(month + 1).padStart(2, "0")}-${String(day).padStart(2, "0")}`;
          const dayTasks = tasks.filter(t => t.dueDate === dateStr || t.startDate === dateStr);
          const isToday = dateStr === "2026-05-24";
          return (
            <div key={day} style={{
              minHeight: 80, background: "#0D0F1A", border: `1px solid ${isToday ? "#6C63FF" : "#1E2130"}`,
              borderRadius: 8, padding: "6px 8px", position: "relative"
            }}>
              <span style={{ color: isToday ? "#6C63FF" : "#666", fontSize: 13, fontWeight: isToday ? 700 : 400 }}>{day}</span>
              {dayTasks.map(t => (
                <div key={t.id} onClick={() => onTaskClick(t)} style={{
                  marginTop: 3, padding: "2px 6px", borderRadius: 4, background: `${t.color}33`,
                  borderLeft: `3px solid ${t.color}`, fontSize: 10, color: "#CCC", cursor: "pointer",
                  whiteSpace: "nowrap", overflow: "hidden", textOverflow: "ellipsis"
                }}>{t.title}</div>
              ))}
            </div>
          );
        })}
      </div>
    </div>
  );
}

      </div>
      <div style={{ padding: 16 }}>
        {overdue.length > 0 && <>
          <div style={{ color: "#FF6584", fontSize: 11, fontWeight: 700, textTransform: "uppercase", letterSpacing: 1, marginBottom: 8 }}>⚠ En retard</div>
          {overdue.map(t => (
            <div key={t.id} style={{ padding: "10px 12px", background: "#1A0D0D", border: "1px solid #FF658433", borderRadius: 10, marginBottom: 8 }}>
              <div style={{ color: "#FF6584", fontSize: 13, fontWeight: 600 }}>{t.title}</div>
              <div style={{ color: "#888", fontSize: 11 }}>Échéance: {formatDate(t.dueDate)}</div>
            </div>
          ))}
        </>}
        {soon.length > 0 && <>
          <div style={{ color: "#FFB347", fontSize: 11, fontWeight: 700, textTransform: "uppercase", letterSpacing: 1, marginBottom: 8, marginTop: 12 }}>⏰ Bientôt</div>
          {soon.map(t => (
            <div key={t.id} style={{ padding: "10px 12px", background: "#1A1500", border: "1px solid #FFB34733", borderRadius: 10, marginBottom: 8 }}>
              <div style={{ color: "#FFB347", fontSize: 13, fontWeight: 600 }}>{t.title}</div>
              <div style={{ color: "#888", fontSize: 11 }}>{daysLeft(t.dueDate)} jour(s) restants</div>
            </div>
          ))}
        </>}
        {reminders.length > 0 && <>
          <div style={{ color: "#6C63FF", fontSize: 11, fontWeight: 700, textTransform: "uppercase", letterSpacing: 1, marginBottom: 8, marginTop: 12 }}>🔔 Rappels programmés</div>
          {reminders.map(r => (
            <div key={r.id} style={{ padding: "10px 12px", background: "#0F0F1A", border: "1px solid #6C63FF33", borderRadius: 10, marginBottom: 8 }}>
              <div style={{ color: "#CCC", fontSize: 13 }}>{r.taskTitle}</div>
              <div style={{ color: "#888", fontSize: 11 }}>{r.label} — {formatDate(r.date)}</div>
            </div>
          ))}
        </>}
        {overdue.length === 0 && soon.length === 0 && reminders.length === 0 && (
          <div style={{ textAlign: "center", color: "#555", padding: 24 }}>Aucune notification</div>
        )}
      </div>
    </div>
  );
}

// ─── MAIN APP ─────────────────────────────────────────────────────────────────
export default function App() {
  const [tasks, setTasks] = useState(INITIAL_TASKS);
  const [view, setView] = useState("kanban");
  const [selectedTask, setSelectedTask] = useState(null);
  const [isNew, setIsNew] = useState(false);
  const [showNotif, setShowNotif] = useState(false);
  const [search, setSearch] = useState("");
  const [filterStatus, setFilterStatus] = useState("all");
  const [filterPriority, setFilterPriority] = useState("all");
  const [notifBadge, setNotifBadge] = useState(true);

  // Browser notifications
  useEffect(() => {
    if ("Notification" in window && Notification.permission === "default") {
      Notification.requestPermission();
    }
    const overdue = tasks.filter(t => t.dueDate && daysLeft(t.dueDate) !== null && daysLeft(t.dueDate) < 0 && t.status !== "done");
    if (overdue.length > 0 && Notification.permission === "granted") {
      new Notification("⚠ Tâches en retard", { body: `${overdue.length} tâche(s) en retard`, icon: "📋" });
    }
  }, []);

  const filteredTasks = tasks.filter(t => {
    if (search && !t.title.toLowerCase().includes(search.toLowerCase()) && !t.assignee.toLowerCase().includes(search.toLowerCase())) return false;
    if (filterStatus !== "all" && t.status !== filterStatus) return false;
    if (filterPriority !== "all" && t.priority !== filterPriority) return false;
    return true;
  });

  const handleSaveTask = (task) => {
    setTasks(prev => prev.some(t => t.id === task.id) ? prev.map(t => t.id === task.id ? task : t) : [...prev, task]);
  };
  const handleDeleteTask = (id) => setTasks(prev => prev.filter(t => t.id !== id));
  const handleAddTask = (status) => {
    setSelectedTask({ status });
    setIsNew(true);
  };
  const handleTaskClick = (task, directUpdate = false) => {
    if (directUpdate) { handleSaveTask(task); return; }
    setSelectedTask(task);
    setIsNew(false);
  };

  const notifCount = tasks.filter(t => (t.dueDate && daysLeft(t.dueDate) !== null && daysLeft(t.dueDate) <= 3 && t.status !== "done") || (t.reminders?.length > 0)).length;

  return (
    <div style={{ minHeight: "100vh", background: "#080A12", fontFamily: "'DM Sans', 'Segoe UI', sans-serif", color: "#F0F0FF" }}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=DM+Sans:wght@300;400;500;600;700;800&display=swap');
        * { box-sizing: border-box; }
        ::-webkit-scrollbar { width: 6px; height: 6px; }
        ::-webkit-scrollbar-track { background: #0D0F1A; }
        ::-webkit-scrollbar-thumb { background: #2A2D3A; border-radius: 3px; }
        input[type=color]::-webkit-color-swatch-wrapper { padding: 0; }
        input[type=color]::-webkit-color-swatch { border: none; border-radius: 6px; }
      `}</style>

      {/* TOPBAR */}
      <div style={{ display: "flex", alignItems: "center", gap: 16, padding: "12px 24px", borderBottom: "1px solid #1E2130", background: "#0D0F1A", position: "sticky", top: 0, zIndex: 100 }}>
        <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
          <div style={{ width: 32, height: 32, borderRadius: 10, background: "linear-gradient(135deg, #6C63FF, #43D9AD)", display: "flex", alignItems: "center", justifyContent: "center", fontSize: 16 }}>⬛</div>
          <span style={{ fontWeight: 800, fontSize: 18, letterSpacing: -0.5, background: "linear-gradient(90deg, #6C63FF, #43D9AD)", WebkitBackgroundClip: "text", WebkitTextFillColor: "transparent" }}>FlowBoard</span>
        </div>

        {/* Search */}
        <div style={{ position: "relative", flex: 1, maxWidth: 320 }}>
          <span style={{ position: "absolute", left: 12, top: "50%", transform: "translateY(-50%)", color: "#444", fontSize: 14 }}>🔍</span>
          <input value={search} onChange={e => setSearch(e.target.value)} placeholder="Rechercher..." style={{ ...inputStyle, paddingLeft: 36, marginTop: 0, height: 36 }} />
        </div>

        {/* Filters */}
        <select value={filterStatus} onChange={e => setFilterStatus(e.target.value)} style={{ ...inputStyle, marginTop: 0, height: 36, width: "auto", paddingRight: 24 }}>
          <option value="all">Tous statuts</option>
          {COLUMNS.map(c => <option key={c.id} value={c.id}>{c.label}</option>)}
        </select>
        <select value={filterPriority} onChange={e => setFilterPriority(e.target.value)} style={{ ...inputStyle, marginTop: 0, height: 36, width: "auto", paddingRight: 24 }}>
          <option value="all">Toutes priorités</option>
          {Object.entries(PRIORITY_LABELS).map(([k, v]) => <option key={k} value={k}>{v}</option>)}
        </select>

        <div style={{ marginLeft: "auto", display: "flex", gap: 10 }}>
          <button onClick={() => { setShowNotif(s => !s); setNotifBadge(false); }} style={{ position: "relative", background: "#1A1D2E", border: "1px solid #2A2D3A", color: "#DDD", padding: "6px 14px", borderRadius: 10, cursor: "pointer", fontSize: 16, fontFamily: "inherit" }}>
            🔔
            {notifBadge && notifCount > 0 && <span style={{ position: "absolute", top: -4, right: -4, background: "#FF6584", color: "#fff", fontSize: 9, fontWeight: 700, width: 16, height: 16, borderRadius: "50%", display: "flex", alignItems: "center", justifyContent: "center" }}>{notifCount}</span>}
          </button>
          <button onClick={() => handleAddTask("todo")} style={{ background: "linear-gradient(135deg, #6C63FF, #4845C8)", border: "none", color: "#fff", padding: "8px 18px", borderRadius: 10, cursor: "pointer", fontSize: 13, fontWeight: 600, fontFamily: "inherit", display: "flex", alignItems: "center", gap: 6 }}>
            + Nouvelle tâche
          </button>
        </div>
      </div>

      {/* VIEW TABS */}
      <div style={{ display: "flex", gap: 4, padding: "12px 24px 0", borderBottom: "1px solid #1E2130", background: "#0D0F1A", overflowX: "auto" }}>
        {VIEWS.map(v => (
          <button key={v.id} onClick={() => setView(v.id)} style={{
            padding: "8px 16px", borderRadius: "10px 10px 0 0", border: "none", cursor: "pointer", fontFamily: "inherit",
            fontSize: 13, fontWeight: view === v.id ? 700 : 400, transition: "all .2s",
            background: view === v.id ? "#080A12" : "transparent",
            color: view === v.id ? "#F0F0FF" : "#666",
            borderBottom: view === v.id ? "2px solid #6C63FF" : "2px solid transparent",
            display: "flex", alignItems: "center", gap: 6, whiteSpace: "nowrap"
          }}>
            {v.icon} {v.label}
          </button>
        ))}
      </div>

      {/* CONTENT */}
      <div>
        {view === "kanban" && <KanbanView tasks={filteredTasks} onTaskClick={handleTaskClick} onAddTask={handleAddTask} />}
        {view === "list" && <ListView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
        {view === "gantt" && <GanttView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
        {view === "timeline" && <TimelineView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
        {view === "calendar" && <CalendarView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
        {view === "mindmap" && <MindMapView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
        {view === "spreadsheet" && <SpreadsheetView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
        {view === "dashboard" && <DashboardView tasks={filteredTasks} onTaskClick={handleTaskClick} />}
      </div>

      {/* MODAL */}
      {selectedTask && (
        <TaskModal
          task={isNew ? null : selectedTask}
          onClose={() => setSelectedTask(null)}
          onSave={task => handleSaveTask(isNew ? { ...task, status: selectedTask.status || "todo" } : task)}
          onDelete={handleDeleteTask}
        />
      )}

      {/* NOTIF PANEL */}
      {showNotif && <NotifPanel tasks={tasks} onClose={() => setShowNotif(false)} />}
    </div>
  );
}
